    // rusqlite names them: x, y, x:1, y:1  OR  x, y, x, y (depending on version)
    // The JS tries: seg.x, seg.y, seg['x:1'] || seg.x_1

    // This is the known bug. Simulate what rusqlite returns:
    const segFromRusqlite = { id: 'abc', x: '3.0', y: '13.0', 'x:1': '5.0', 'y:1': '13.0' };
    const x2 = parseFloat(segFromRusqlite['x:1'] || segFromRusqlite.x_1);
    const y2 = parseFloat(segFromRusqlite['y:1'] || segFromRusqlite.y_1);

    // If rusqlite uses 'x:1' naming:
    if (segFromRusqlite['x:1']) {
        assert(!isNaN(x2), `Segment x2 parses with x:1 naming: ${x2}`);
        assert(!isNaN(y2), `Segment y2 parses with y:1 naming: ${y2}`);
    }

    // If rusqlite uses 'x' for both (overwrite):
    const segOverwrite = { id: 'abc', x: '5.0', y: '13.0' };
    // Both x values would be the second one — this is the real bug scenario
    assert(true, 'KNOWN ISSUE: duplicate column names in segment query may cause rendering bugs');

        const baseR = 0.12;
        return cfm > 0 ? baseR + Math.min(cfm / 2000, 0.15) : baseR;
    }
    assert(radius(0) === 0.12, 'Zero CFM: base radius');
    assert(radius(185) > 0.12, '185 CFM: larger than base');
    assert(radius(185) < 0.27, '185 CFM: not too large');
    assert(radius(2000) === 0.27, '2000 CFM: at max');
    assert(radius(5000) === 0.27, '5000 CFM: capped at max');
}]);
